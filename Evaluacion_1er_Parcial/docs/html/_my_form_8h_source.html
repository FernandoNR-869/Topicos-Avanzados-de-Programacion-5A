70 void AplicarFiltros() {
71 if (frameOriginal->empty()) return;
72
73 int val = tkUmbral->Value;
74 String^ opcion = cbFiltros->Text;
75
76 if (opcion == "Escala de grises") {
77 cv::Mat gris;
78 cv::cvtColor(*frameOriginal, gris, cv::COLOR_BGR2GRAY);
79 gris.convertTo(*frameProcesado, -1, 1, val - 128);
80 }
81 else if (opcion == "Binarizacin (B/N)") {
82 cv::cvtColor(*frameOriginal, *frameProcesado, cv::COLOR_BGR2GRAY);
83 cv::threshold(*frameProcesado, *frameProcesado, val, 255, cv::THRESH_BINARY);
84 }
85 else if (opcion == "Binarizacin (N/B)") {
86 cv::cvtColor(*frameOriginal, *frameProcesado, cv::COLOR_BGR2GRAY);
87 cv::threshold(*frameProcesado, *frameProcesado, val, 255, cv::THRESH_BINARY_INV);
88 }
89 else if (opcion == "Cany") {
90 cv::Canny(*frameOriginal, *frameProcesado, val, val * 3);
91 }
92 else if (opcion == "Sobel") {
93 cv::Mat temp, gx, gy, agx, agy;
94 cv::cvtColor(*frameOriginal, temp, cv::COLOR_BGR2GRAY);
95 cv::Sobel(temp, gx, CV_16S, 1, 0, 3);
96 cv::Sobel(temp, gy, CV_16S, 0, 1, 3);
97 cv::convertScaleAbs(gx, agx); cv::convertScaleAbs(gy, agy);
98 cv::addWeighted(agx, 0.5, agy, 0.5, 0, *frameProcesado);
99 cv::threshold(*frameProcesado, *frameProcesado, val, 255, cv::THRESH_TOZERO);
100 }
101 else if (opcion == "Separacin RGB") {
102 std::vector<cv::Mat> canales; cv::split(*frameOriginal, canales);
103 cv::Mat vacio = cv::Mat::zeros(frameOriginal->size(), CV_8UC1);
104 std::vector<cv::Mat> mezcla;
105 if (val <= 85) { mezcla.push_back(canales[0]); mezcla.push_back(vacio); mezcla.push_back(vacio); }
106 else if (val <= 170) { mezcla.push_back(vacio); mezcla.push_back(canales[1]); mezcla.push_back(vacio); }
107 else { mezcla.push_back(vacio); mezcla.push_back(vacio); mezcla.push_back(canales[2]); }
108 cv::merge(mezcla, *frameProcesado);
109 }
110 else if (opcion == "Rotacin") {
111 cv::Point2f c(frameOriginal->cols / 2.0f, frameOriginal->rows / 2.0f);
112 cv::Mat r = cv::getRotationMatrix2D(c, (val * 360 / 255.0), 1.0);
113 cv::warpAffine(*frameOriginal, *frameProcesado, r, frameOriginal->size());
114 }
115 else if (opcion == "Zoom") {
116 double s = 1.0 + (val / 127.0);
117 cv::Mat temp; cv::resize(*frameOriginal, temp, cv::Size(), s, s, cv::INTER_LINEAR);
118 int x = (temp.cols - frameOriginal->cols) / 2;
119 int y = (temp.rows - frameOriginal->rows) / 2;
120 cv::Rect roi(x, y, frameOriginal->cols, frameOriginal->rows);
121 *frameProcesado = temp(roi).clone();
122 }
123 else {
124 *frameProcesado = frameOriginal->clone();
125 }
126 ActualizarPantalla(*frameProcesado);
127 }
